**`tma/src/components/auth/LoginForm.tsx`** — Client Component:
```tsx
"use client"

import { useActionState } from "react"
import { signIn } from "@/auth"
import { Input } from "@/components/ui/input"
import { Label } from "@/components/ui/label"
import { TextureButton } from "@/components/ui/texture-button"
import { motion } from "motion/react"
import Link from "next/link"
import { AuthError } from "next-auth"

async function loginAction(
  _: { error?: string } | undefined,
  formData: FormData
): Promise<{ error?: string } | undefined> {
  "use server"
  try {
    await signIn("credentials", {
      email: formData.get("email") as string,
      password: formData.get("password") as string,
      redirectTo: "/tma",
    })
  } catch (error) {
    if (error instanceof AuthError) {
      return { error: "Email o contraseña incorrectos" }
    }
    throw error // Re-lanzar NEXT_REDIRECT
  }
}

export function LoginForm() {
  const [state, action, isPending] = useActionState(loginAction, undefined)

  return (
    <motion.div
      initial={{ opacity: 0, y: 16 }}
      animate={{ opacity: 1, y: 0 }}
      transition={{ duration: 0.35, ease: "easeOut" }}
      className="w-full max-w-[400px] mx-auto px-4"
    >
      <h1 className="text-[28px] font-bold text-brand-title mb-8 text-center">
        Iniciar sesión
      </h1>

      <form action={action} className="flex flex-col gap-5">
        <div className="flex flex-col gap-1.5">
          <Label htmlFor="email" className="text-[14px] font-normal text-brand-text">
            Email
          </Label>
          <Input
            id="email"
            name="email"
            type="email"
            placeholder="tu@nyc.com.ar"
            required
            autoComplete="email"
            className="bg-white border-brand-accent focus-visible:ring-brand-primary"
          />
        </div>

        <div className="flex flex-col gap-1.5">
          <Label htmlFor="password" className="text-[14px] font-normal text-brand-text">
            Contraseña
          </Label>
          <Input
            id="password"
            name="password"
            type="password"
            placeholder="••••••••"
            required
            autoComplete="current-password"
            className="bg-white border-brand-accent focus-visible:ring-brand-primary"
          />
        </div>

        {state?.error && (
          <p className="text-[11px] text-red-600" role="alert">
            {state.error}
          </p>
        )}

        <TextureButton
          type="submit"
          disabled={isPending}
          className="w-full bg-brand-primary hover:bg-brand-title text-white font-medium h-11"
        >
          {isPending ? "Ingresando..." : "Iniciar sesión"}
        </TextureButton>
      </form>

      <p className="text-[11px] text-brand-text text-center mt-6">
        ¿No tenés cuenta?{" "}
        <Link href="/register" className="text-brand-primary hover:text-brand-title transition-colors">
          Registrate
        </Link>
      </p>
    </motion.div>
  )
}
```

NOTA IMPORTANTE: Si TextureButton no acepta las clases de override via `className`, leer `src/components/ui/texture-button.tsx` para ver cómo aplicar el color primario (#EA580C). Ajustar según la API real del componente.

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser → LoginForm | FormData con email y password — entrada no confiable |
| Client Component → Server Action | useActionState pasa FormData al Server Action |
| Server Component → auth() | TmaPage verifica sesión antes de renderizar contenido |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-01 | Elevation of Privilege | TmaPage sin sesión | mitigate | `await auth()` + `redirect("/login")` como segunda línea de defensa detrás del middleware — doble verificación de sesión |
| T-03-02 | Tampering | LoginForm Server Action inline | mitigate | Server Action con `"use server"` directive — Next.js 15 aplica protección CSRF automática; el formulario no acepta inputs arbitrarios fuera de email/password |
| T-03-03 | Information Disclosure | Error messages en formularios | accept | Mensajes de error genéricos en login ("Email o contraseña incorrectos") no revelan si el email existe — consistente con la mitigación de timing en auth.ts |
  </threat_model>
