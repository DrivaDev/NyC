### Task 1.5 — Create Clerk middleware

<read_first>
- (no existing middleware.ts — create from scratch at project root)
</read_first>

<action>
Create `menu-digital/middleware.ts` at the project root (same level as `app/`, `package.json`):

```typescript
import { clerkMiddleware, createRouteMatcher } from '@clerk/nextjs/server'

const isProtectedRoute = createRouteMatcher(['/dashboard(.*)'])

export default clerkMiddleware(async (auth, req) => {
  if (isProtectedRoute(req)) {
    await auth.protect()
  }
})

export const config = {
  matcher: [
    '/((?!_next|[^?]*\\.(?:html?|css|js(?!on)|jpe?g|webp|png|gif|svg|ttf|woff2?|ico|csv|docx?|xlsx?|zip|webmanifest)).*)',
    '/(api|trpc)(.*)',
  ],
}
```

Critical: this file MUST be at `menu-digital/middleware.ts`, not inside `app/`.
</action>
